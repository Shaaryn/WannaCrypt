33  public byte[] Decrypt(TextDecryptionObject decryptObject, int offset)
34  {
35  // Copy 16 bits of the encripted message to the state.
36  this.state = new byte[BlockSize];
37  for (int i = 0; i < BlockSize; i++)
38  {
39  state[i] = decryptObject.EncryptedMessageCiphers[offset][i];
40  }
41 
42  // 8 because in this case : 16 * (9+1) = 144 so the key will be used from top to bottom
43  AddRoundKey(decryptObject.ExpandedKey, (int)decryptObject.Round);
44 
45  for (int i = (int)decryptObject.Round - 1; i >= 0; i--)
46  {
47  InvShiftRows();
48  InvSubBytes();
49  AddRoundKey(decryptObject.ExpandedKey, i);
50  InvMixColumns();
51  }
52 
53  // Final round
54  InvShiftRows();
55  InvSubBytes();
56  AddRoundKey(decryptObject.Key, -1);
57 
58  return this.state;
59  }
