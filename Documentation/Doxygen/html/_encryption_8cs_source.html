33  public byte[] Encrypt(TextEncryptionObject encryptObject, int offset)
34  {
35  // Copy 16 bits of the message to the state.
36  this.state = new byte[BlockSize];
37  for (int i = 0; i < BlockSize; i++)
38  {
39  state[i] = encryptObject.PaddedMessage[i + offset];
40  }
41 
42  // -1 because in this case : 16 * 0 = 0 so the currentKey index is not incremented in the function.
43  AddRoundKey(encryptObject.Key, -1);
44 
45  for (int i = 0; i < (int)encryptObject.Round; i++)
46  {
47  SubBytes();

49  MixColumns();
50  AddRoundKey(encryptObject.ExpandedKey, i);
51  }
52

55  ShiftRows();
56  AddRoundKey(encryptObject.ExpandedKey, (int)encryptObject.Round);
