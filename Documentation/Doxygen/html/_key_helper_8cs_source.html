27  public byte[] KeyExpansion(IAESObject aesObject)
28  {
29  // 128 -> 10 * 16 +16
30  // 10 : We need a 16 bytes key for each round
31  // 16(at the end): we add the original key as well.
32  int keySize = (((int)aesObject.Round + 1) * 16) + 16;
33 
34  byte[] expandedKey = new byte[keySize];
35 
36  // Copy the original key as it is.
37  for (int i = 0; i < aesObject.Key.Length; i++)
38  {
39  expandedKey[i] = aesObject.Key[i];
40  }
41 
42  int generatedBytes = aesObject.Key.Length;
43  int rConIndex = 1;
44  byte[] tmpCore = new byte[4];
45 
46  do
47  {
48  // Read the recently generated 4 bytes for the core.
49  for (int i = 0; i < tmpCore.Length; i++)
50  {
51  tmpCore[i] = expandedKey[i + generatedBytes - 4];
52  }
53 
54  // Perform the core once for each key.Length()-th byte key.
55  if (generatedBytes % aesObject.Key.Length == 0)
56  {
57  KeyExpansionCore(tmpCore, rConIndex);
58  rConIndex++;
59  }
60  else if (aesObject.Key.Length == 32 && generatedBytes % aesObject.Key.Length == 16)
61  {
62  tmpCore[0] = Tables.SBox[tmpCore[0]];
63  tmpCore[1] = Tables.SBox[tmpCore[1]];
64  tmpCore[2] = Tables.SBox[tmpCore[2]];
65  tmpCore[3] = Tables.SBox[tmpCore[3]];
66  }
67 
68  for (int i = 0; i < tmpCore.Length; i++)
69  {
70  expandedKey[generatedBytes] = (byte)(expandedKey[generatedBytes - aesObject.Key.Length] ^ tmpCore[i]);
71  generatedBytes++;
72  }
73  }
74  while (generatedBytes < keySize);
75 
76  return expandedKey;
77  }

84  public void KeyExpansionCore(byte[] inputBits, int index)
85  {
86  // Left Rotation

88  inputBits[0] = inputBits[1];
89  inputBits[1] = inputBits[2];
90  inputBits[2] = inputBits[3];
91  inputBits[3] = tmp;
92 
93  // SBox lookup
94  inputBits[0] = Tables.SBox[inputBits[0]];
95  inputBits[1] = Tables.SBox[inputBits[1]];
96  inputBits[2] = Tables.SBox[inputBits[2]];
97  inputBits[3] = Tables.SBox[inputBits[3]];
98 
99  // RCon
