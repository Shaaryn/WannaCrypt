60  public void ExecuteEncryption(TextEncryptionObject encryptObject)
61  {
62  // Define the rounds
63  encryptObject.Round = DefineRounds(encryptObject);
64 
65  // Bytify the text
66  encryptObject.MessageByte = new byte[encryptObject.Message.Length];
67  for (int i = 0; i < encryptObject.Message.Length; i++)
68  {
69  encryptObject.MessageByte[i] = Convert.ToByte(encryptObject.Message[i]);
70  }
71 
72  // Pad the message
73  encryptObject.PaddedMessage = helpMeFormat.CalculatePadding(encryptObject);
74 
75  // Calculate the expanded key
76  encryptObject.Key = new byte[(int)encryptObject.Size];
77  int n = 0;
78  if (encryptObject.KeyString != null)
79  {
80  for (int i = 0; i < encryptObject.KeyString.Length; i += 3)
81  {
82  encryptObject.Key[n] = Convert.ToByte(encryptObject.KeyString.Substring(i, 2), 16);
83  n++;
84  }
85  }
86 
87  encryptObject.ExpandedKey = helpMeKey.KeyExpansion(encryptObject);
88 
89  // Execute the encryption process for each block
90  Encryption encryption = new Encryption();
91  byte[] tmpEncryptedMessage = new byte[BlockSize];
92  encryptObject.EncryptedMessage = new byte[encryptObject.PaddedMessage.Length];
93 
94  for (int i = 0; i < encryptObject.PaddedMessage.Length; i += BlockSize)
95  {
96  tmpEncryptedMessage = encryption.Encrypt(encryptObject, i);
97  for (int j = 0; j < BlockSize; j++)
98  {
99  encryptObject.EncryptedMessage[j + i] = tmpEncryptedMessage[j];
100  }
101  }
102 
103  encryptObject.EncryptedMessageString = BitConverter.ToString(encryptObject.EncryptedMessage);
104  }

110  public void ExecuteDecryption(TextDecryptionObject decryptObject)
111  {
112  // Defines the rounds
113  decryptObject.Round = DefineRounds(decryptObject);
114 
115  // Break the message into blocksized arrays
116  decryptObject.EncryptedMessageCiphers = new List<byte[]>();
117  decryptObject.EncryptedMessageCiphers = helpMeFormat.CalculateArrays(decryptObject);
118 
119  // Calculate the expanded key.
120  decryptObject.Key = new byte[(int)decryptObject.Size];
121  int k = 0;
122  for (int i = 0; i < decryptObject.KeyString.Length; i += 3)
123  {
124  decryptObject.Key[k] = Convert.ToByte(decryptObject.KeyString.Substring(i, 2), 16);
125  k++;
126  }
127 
128  decryptObject.ExpandedKey = helpMeKey.KeyExpansion(decryptObject);
129 
130  // Execute the decryption process for each block
131  Decryption decryption = new Decryption();
132  byte[] tmpDecryptedMessageByte = new byte[BlockSize];
133  decryptObject.MessageByte = new byte[BlockSize * decryptObject.EncryptedMessageCiphers.Count];
134 
135  for (int i = 0; i < decryptObject.EncryptedMessageCiphers.Count; i++)
136  {
137  tmpDecryptedMessageByte = decryption.Decrypt(decryptObject, i);
138  for (int j = 0; j < BlockSize; j++)
139  {
140  decryptObject.MessageByte[(i * BlockSize) + j] = tmpDecryptedMessageByte[j];
141  }
142  }
143 
144  decryptObject.Message = Encoding.UTF8.GetString(decryptObject.MessageByte);
145  }
